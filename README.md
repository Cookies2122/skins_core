# skins_core


# RU
mm_giveskin <steamid64> <skin_id> [seconds]    // 0 или без аргумента = навсегда
mm_removeskin <steamid64> <skin_id>            // убрать конкретный скин из БД
mm_revokeaccess <steamid64>                    // убрать все скины из БД
mm_listskinscore                               // список всех skin_id из skins.ini
mm_reloadskins                                 // перезагрузить конфиг/скины/переводы

# EN
mm_giveskin <steamid64> <skin_id> [seconds]    // Give skin. 0 or omitted = permanent
mm_removeskin <steamid64> <skin_id>            // Remove specific skin from DB
mm_revokeaccess <steamid64>                    // Remove all skins from DB
mm_listskinscore                               // List all skin IDs from skins.ini
mm_reloadskins                                 // Reload config / skins / translations
